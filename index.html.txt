/* ======= UTIL ======= */
function formatPrice(v){ return "R$ " + v.toFixed(2).replace('.',','); }

/* ======= RENDER PRODUCTS ======= */
const grid = document.getElementById('grid');
function renderProducts(){
  grid.innerHTML = '';
  products.forEach(p=>{
    const card = document.createElement('div'); card.className='card';
    card.innerHTML = `
      <img src="${p.image}" alt="${p.name}">
      <h3>${p.name}</h3>
      <div class="price">${formatPrice(p.price)}</div>
      <div class="actions">
        <a class="view" href="javascript:void(0)" onclick="openProduct(${p.id})">Ver</a>
        <button class="add" onclick="addToCart(${p.id})">Adicionar</button>
      </div>
    `;
    grid.appendChild(card);
  });
}

/* ======= CART (localStorage) ======= */
function getCart(){ try{ return JSON.parse(localStorage.getItem('md_cart')||'[]'); }catch(e){return []} }
function saveCart(c){ localStorage.setItem('md_cart', JSON.stringify(c)); renderCartCount(); renderCartItems(); }
function addToCart(id, qty=1){
  const prod = products.find(x=>x.id===id); if(!prod) return alert('Produto não encontrado');
  const cart = getCart(); const item = cart.find(i=>i.id===id);
  if(item) item.qty += qty; else cart.push({ id:prod.id, name:prod.name, price:prod.price, image:prod.image, qty:qty });
  saveCart(cart);
  toast('Produto adicionado ao carrinho');
}
function removeFromCart(id){ const cart = getCart().filter(i=>i.id!==id); saveCart(cart); }
function changeQty(id, delta){ const cart = getCart(); const it = cart.find(i=>i.id===id); if(!it) return; it.qty += delta; if(it.qty<1) it.qty=1; saveCart(cart); }
function cartTotal(){ return getCart().reduce((s,i)=>s + (i.price*i.qty), 0); }

function renderCartCount(){
  const count = getCart().reduce((s,i)=>s+i.qty,0); document.getElementById('cart-count').innerText = count;
}

/* ======= RENDER CART UI ======= */
function renderCartItems(){
  const container = document.getElementById('cart-items');
  if(!container) return;
  const c = getCart();
  container.innerHTML = '';
  if(c.length===0){ container.innerHTML = '<p>Seu carrinho está vazio.</p>'; document.getElementById('cart-total').innerText = formatPrice(0); renderCartCount(); return; }
  c.forEach(it=>{
    const div = document.createElement('div'); div.className='cart-item';
    div.innerHTML = `
      <img src="${it.image}" alt="${it.name}">
      <div style="flex:1">
        <div style="font-weight:700">${it.name}</div>
        <div style="color:var(--muted)">${formatPrice(it.price)} • Qtd: ${it.qty}</div>
        <div style="margin-top:8px">
          <button class="btn" onclick="changeQty(${it.id},-1)">-</button>
          <button class="btn" onclick="changeQty(${it.id},1)">+</button>
          <button class="btn" onclick="removeFromCart(${it.id})">Remover</button>
        </div>
      </div>
    `;
    container.appendChild(div);
  });
  document.getElementById('cart-total').innerText = formatPrice(cartTotal());
  renderCartCount();
}

/* ======= CART MODAL CONTROLS ======= */
const cartModal = document.getElementById('cart-modal');
document.getElementById('open-cart').addEventListener('click', ()=>{ cartModal.classList.remove('hidden'); cartModal.setAttribute('aria-hidden','false'); renderCartItems(); });
document.getElementById('close-cart').addEventListener('click', ()=>{ cartModal.classList.add('hidden'); cartModal.setAttribute('aria-hidden','true'); });

document.getElementById('clear-cart').addEventListener('click', ()=>{ saveCart([]); });

document.getElementById('checkout').addEventListener('click', ()=> {
  const c = getCart(); if(c.length===0) return alert('Seu carrinho está vazio.');
  const total = cartTotal();
  // Simulated payment flow
  alert('Compra simulada concluída!\nTotal: ' + formatPrice(total) + '\nObrigado pela preferência.');
  saveCart([]);
  cartModal.classList.add('hidden');
});

/* ======= PRODUCT DETAIL MODAL ======= */
const productModal = document.getElementById('product-modal');
function openProduct(id){
  const p = products.find(x=>x.id===id); if(!p) return;
  const container = document.getElementById('product-detail');
  container.innerHTML = `
    <div style="flex:1;min-width:280px">
      <img src="${p.image}" alt="${p.name}" style="width:100%;border-radius:8px">
    </div>
    <div style="flex:1;min-width:260px">
      <h2 style="color:var(--gold)">${p.name}</h2>
      <p style="color:var(--muted)">${p.description}</p>
      <div style="font-weight:800;font-size:20px;margin:10px 0">${formatPrice(p.price)}</div>
      <div style="display:flex;gap:10px">
        <button class="btn primary" onclick="buyNow(${p.id})">Comprar agora</button>
        <button class="btn" onclick="addToCart(${p.id})">Adicionar ao carrinho</button>
      </div>
    </div>
  `;
  productModal.classList.remove('hidden'); productModal.setAttribute('aria-hidden','false');
}
document.getElementById('close-product').addEventListener('click', ()=>{ productModal.classList.add('hidden'); productModal.setAttribute('aria-hidden','true'); });

/* Buy now: add + checkout simulated */
function buyNow(id){ addToCart(id,1); setTimeout(()=>{ document.getElementById('checkout').click(); }, 200); }

/* Small toast helper */
function toast(msg){ try{ if(window.navigator && window.navigator.vibrate) window.navigator.vibrate(30); }catch(e){} alert(msg); }
